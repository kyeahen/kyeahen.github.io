public class BJ_2261 {

    public static void main(String[] args) throws NumberFormatException, IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st;

        int N = Integer.parseInt(br.readLine());
        ArrayList<Point> list = new ArrayList<>();

        for (int i = 0; i < N; i++) {
            st = new StringTokenizer(br.readLine());

            int x = Integer.parseInt(st.nextToken());
            int y = Integer.parseInt(st.nextToken());

    public static int dist(Point p1, Point p2) {
        int x = p1.x - p2.x;
        int y = p1.y - p2.y;

        return (x * x) + (y * y);
    }

    // ì™„ì „ íƒìƒ‰ìœ¼ë¡œ ê°€ì¥ ê°€ê¹Œìš´ ê±°ë¦¬ ì°¾ê¸°
    static int bruteForce(ArrayList<Point> list, int start, int end) {
        int min = Integer.MAX_VALUE;

        for (int i = start; i < end; i++) {
            for (int j = i + 1; j <= end; j++) {

                int diff = dist(list.get(i), list.get(j));
                min = Math.min(diff, min);
            }
        }

        return min;
    }

    public static int solution(ArrayList<Point> arrList, int start, int end) {
        int n = end - start + 1; //ì¢Œí‘œ ê°œìˆ˜

        // ì¢Œí‘œ ê°œìˆ˜ê°€ 3 ì´í•˜ë©´ ì™„ì „íƒìƒ‰ìœ¼ë¡œ ê°€ì¥ ê°€ê¹Œìš´ ë‘ ì  ì‚¬ì´ì˜ ê±°ë¦¬ë¥¼ ì°¾ê¸°
        if (n <= 3) {

        }

        int mid = (start + end) / 2;

        // ì¤‘ì•™ì„ ì„ ê¸°ì¤€ìœ¼ë¡œ ì™¼ìª½ ì ë“¤ ì¤‘ ê°€ì¥ ì‘ì€ ê±°ë¦¬ = d1 (1. ê°€ì¥ ê°€ê¹Œìš´ ë‘ ì ì´ ì™¼ìª½ ì˜ì—­ì— ìˆëŠ” ê²½ìš°)
        // ì˜¤ë¥¸ìª½ ì ë“¤ ì¤‘ ê°€ì¥ ì‘ì€ ê±°ë¦¬ = d2 (2. ê°€ì¥ ê°€ê¹Œìš´ ë‘ ì ì´ ì˜¤ë¥¸ìª½ ì˜ì—­ì— ìˆëŠ” ê²½ìš°)
        // min(d1, d2)ë¥¼ êµ¬í•˜ë©´ ëœë‹¤.
        int d = Math.min(solution(arrList, start, mid), solution(arrList, mid + 1, end));

        //3. ê°€ì¥ ê°€ê¹Œìš´ ë‘ ì ì´ ì™¼ìª½ê³¼ ì˜¤ë¥¸ìª½ ì˜ì—­ í•˜ë‚˜ì”© ìˆëŠ” ê²½ìš° : ì¤‘ì•™ì„ ì„ ê¸°ì ìœ¼ë¡œ dë§Œí¼ ë–¨ì–´ì§„ ì ë§Œ ë³¸ë‹¤.
        // - ì¤‘ì•™ì„ ì„ ê¸°ì¤€ìœ¼ë¡œ ì–‘ìª½ìœ¼ë¡œ d ê±°ë¦¬ ì´ë‚´ì— ë“¤ì–´ì˜¤ëŠ” ì ë“¤ë§Œ ë³´ë©´ ë¨.
        ArrayList<Point> band = new ArrayList<>();

        for (int i = start; i <= end; i++) {
            int tx = arrList.get(mid).x - arrList.get(i).x;

            if (tx * tx < d) { // d ë¯¸ë§Œì´ë©´ ì¶”ê°€
                band.add(arrList.get(i));
            }
        }

        Collections.sort(band, (p1, p2) -> Integer.compare(p1.y, p2.y)); //yì¢Œí‘œ ê¸°ì¤€ ì˜¤ë¦„ì°¨ìˆœ ì •ë ¬

        // yì¢Œí‘œë¥¼ ê¸°ì¤€ìœ¼ë¡œ ì˜¤ë¦„ì°¨ìˆœ ì •ë ¬ëœ bandì˜ ê° ìš”ì†Œ(P1, P2, ...)ëŠ”
        // ìê¸°ë³´ë‹¤ í° ìš”ì†Œë§Œ ë³´ë©´ì„œ ê±°ë¦¬ë¥¼ ì¸¡ì •í•¨.
        for (int i = 0; i < band.size() - 1; i++) {
            for (int j = i + 1; j < band.size(); j++) {
                int ty = band.get(j).y - band.get(i).y;

                if (ty * ty < d) { //d ë¯¸ë§Œ
                    d = Math.min(d, dist(band.get(i), band.get(j)));

                    break;

    }
